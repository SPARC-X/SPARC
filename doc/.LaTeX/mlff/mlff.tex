\begin{frame}[allowframebreaks]{\texttt{MLFF\_FLAG}} \label{MLFF_FLAG}
\vspace*{-12pt}
\begin{columns}
\column{0.4\linewidth}
\begin{block}{Type}
Integer
\end{block}

\begin{block}{Default}
0
\end{block}

\column{0.4\linewidth}
\begin{block}{Unit}
No unit
\end{block}

\begin{block}{Example}
\texttt{MLFF\_FLAG}: 1
\end{block}
\end{columns}

\begin{block}{Description}
Flag to turn on MLFF in SPARC. There are three options currently available. (1) MLFF\_FLAG: 1 to perform on-the-fly MD with no existing model, (2) MLFF\_FLAG: 21 to perform only predictions from an existing model, and (3) MLFF\_FLAG: 22 to perform on-the-fly MD building on top of an existing model
\end{block}

\end{frame}

\begin{frame}[allowframebreaks]{\texttt{{MLFF\_REGUL\_MIN}}} \label{MLFF_REGUL_MIN}
\vspace*{-12pt}
\begin{columns}
\column{0.4\linewidth}
\begin{block}{Type}
Double
\end{block}

\begin{block}{Default}
1e-14
\end{block}

\column{0.4\linewidth}
\begin{block}{Unit}
No unit
\end{block}

\begin{block}{Example}
\texttt{MLFF\_REGUL\_MIN}: 1e-10
\end{block}
\end{columns}

\begin{block}{Description}
 The mimimum value of inverse of condition number of $K^TK + \lambda I$ matrix which appear in MLFF calculation.
\end{block}

\begin{block}{Remark}
The $K^TK + \lambda I$ matrix needs to be inverted during the training of MLFF. The matrix $K^TK$ is in general ill-conditioned so regularization is used to improve the conditioning. Any number in the range of 1E-10-1E-14 should work. A larger value should result in lesser DFT calls, but also lesser accuracy of MLFF model.
\end{block}
\end{frame}

\begin{frame}[allowframebreaks]{\texttt{{MLFF\_FACTOR\_MULTIPLY\_SIGMATOL}}} \label{MLFF_FACTOR_MULTIPLY_SIGMATOL}
\vspace*{-12pt}
\begin{columns}
\column{0.4\linewidth}
\begin{block}{Type}
Double
\end{block}

\begin{block}{Default}
1.01
\end{block}

\column{0.4\linewidth}
\begin{block}{Unit}
No unit
\end{block}

\begin{block}{Example}
\texttt{MLFF\_FACTOR\_MULTIPLY\_SIGMATOL}: 1.001
\end{block}
\end{columns}

\begin{block}{Description}
 A prefactor multiplied with maximum of Bayesian errors while updating the $\sigma_{\rm tol}$. It has to be greater than 1.
\end{block}

\begin{block}{Remark}
A larger value will result lesser DFT calls hence might reduce the accuracy. Use any number in (1.0001-1.01) range.
\end{block}
\end{frame}
